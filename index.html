// Colunas das respostas (ajuste conforme sua planilha) // Col1=Timestamp, Col2=CNPJ, Col3=Setor, Col4=Função, Col5=Tempo empresa // Col6-Col57 = 52 perguntas dos 13 eixos (4 por eixo) var COL_CNPJ = 2; var COL_SETOR = 3; var COL_INICIO = 6; // Primeira pergunta // Eixos (índice base 0 relativo à col de início) var EIXOS = [ { num: 1, nome: 'Assédio no trabalho', inicio: 0, fim: 3 }, { num: 2, nome: 'Suporte e apoio no trabalho', inicio: 4, fim: 7 }, { num: 3, nome: 'Gestão de mudanças organizacionais', inicio: 8, fim: 11 }, { num: 4, nome: 'Clareza de papel e função', inicio: 12, fim: 15 }, { num: 5, nome: 'Recompensas e reconhecimento', inicio: 16, fim: 19 }, { num: 6, nome: 'Autonomia e controle no trabalho', inicio: 20, fim: 23 }, { num: 7, nome: 'Justiça organizacional', inicio: 24, fim: 27 }, { num: 8, nome: 'Segurança e ausência de eventos traumáticos', inicio: 28, fim: 31 }, { num: 9, nome: 'Nível adequado de demanda (subcarga)', inicio: 32, fim: 35 }, { num: 10, nome: 'Equilíbrio de demandas (sobrecarga)', inicio: 36, fim: 39 }, { num: 11, nome: 'Qualidade dos relacionamentos', inicio: 40, fim: 43 }, { num: 12, nome: 'Qualidade da comunicação', inicio: 44, fim: 47 }, { num: 13, nome: 'Trabalho remoto e conexão com equipe', inicio: 48, fim: 51 }, ]; // ── Classificação de gravidade (escala 1-5, perguntas no positivo) ── // IMPORTANTE: Como as perguntas são no POSITIVO, médias ALTAS = BOM // Para o diagnóstico, invertemos: score_risco = 6 - média // Score risco 1.00-2.33 = Baixo risco // Score risco 2.34-3.66 = Médio risco // Score risco 3.67-5.00 = Alto risco function classificarGravidade(media) { var score_risco = 6 - media; // Inverte a escala if (score_risco <= 2.33) return 'Baixa'; if (score_risco <= 3.66) return 'Média'; return 'Alta'; } function corGravidade(gravidade) { if (gravidade === 'Baixa') return '#C8E6C9'; // Verde claro if (gravidade === 'Média') return '#FFF9C4'; // Amarelo claro return '#FFCDD2'; // Vermelho claro } // ── Calcular médias ───────────────────────────────────────────── function calcularMedias() { var ss = SpreadsheetApp.openById(PLANILHA_ID); var sheetRespostas = ss.getSheets()[0]; var dados = sheetRespostas.getDataRange().getValues(); if (dados.length < 2) { SpreadsheetApp.getUi().alert('Nenhuma resposta encontrada!'); return; } // Agrupar respostas por CNPJ e Setor var grupos = {}; for (var i = 1; i < dados.length; i++) { var linha = dados[i]; var cnpj = String(linha[COL_CNPJ - 1] || '').trim(); var setor = String(linha[COL_SETOR - 1] || '').trim(); var chave = cnpj + '||' + setor; if (!grupos[chave]) { grupos[chave] = { cnpj: cnpj, setor: setor, respostas: [] }; } grupos[chave].respostas.push(linha); } // Criar ou limpar aba Médias var sheetMedias = ss.getSheetByName('Médias DRPS v3'); if (sheetMedias) { sheetMedias.clearContents(); sheetMedias.clearFormats(); } else { sheetMedias = ss.insertSheet('Médias DRPS v3'); } // Cabeçalho var cabecalho = ['CNPJ', 'Setor', 'Respondentes']; EIXOS.forEach(function(e) { cabecalho.push('E' + e.num + ' — Média'); cabecalho.push('E' + e.num + ' — Score Risco'); cabecalho.push('E' + e.num + ' — Gravidade'); cabecalho.push('E' + e.num + ' — Probabilidade'); cabecalho.push('E' + e.num + ' — Risco Final'); }); sheetMedias.appendRow(cabecalho); // Formatar cabeçalho var rangeCab = sheetMedias.getRange(1, 1, 1, cabecalho.length); rangeCab.setBackground('#1A3A5C'); rangeCab.setFontColor('#FFFFFF'); rangeCab.setFontWeight('bold'); // Calcular e preencher médias var linha_idx = 2; Object.keys(grupos).forEach(function(chave) { var grupo = grupos[chave]; var n = grupo.respostas.length; var linha_dados = [grupo.cnpj, grupo.setor, n]; EIXOS.forEach(function(eixo) { // Calcular média das 4 perguntas do eixo var soma = 0; var count = 0; grupo.respostas.forEach(function(resp) { for (var p = eixo.inicio; p <= eixo.fim; p++) { var val = parseFloat(resp[COL_INICIO - 1 + p]); if (!isNaN(val) && val >= 1 && val <= 5) { soma += val; count++; } } }); var media = count > 0 ? soma / count : 0; var score_risco = media > 0 ? 6 - media : 0; var gravidade = media > 0 ? classificarGravidade(media) : '—'; linha_dados.push(media.toFixed(2)); linha_dados.push(score_risco.toFixed(2)); linha_dados.push(gravidade); linha_dados.push(''); // Probabilidade — preencher manualmente linha_dados.push(''); // Risco Final — calculado pela fórmula abaixo }); sheetMedias.appendRow(linha_dados); // Colorir células de gravidade EIXOS.forEach(function(eixo, idx) { var colGrav = 4 + (idx * 5); // Coluna de gravidade var cellGrav = sheetMedias.getRange(linha_idx, colGrav); var grav = linha_dados[3 + (idx * 5)]; cellGrav.setBackground(corGravidade(grav)); // Adicionar validação de dados para probabilidade var colProb = 5 + (idx * 5); var cellProb = sheetMedias.getRange(linha_idx, colProb); var regra = SpreadsheetApp.newDataValidation() .requireValueInList(['Baixa', 'Média', 'Alta']) .setAllowInvalid(false) .build(); cellProb.setDataValidation(regra); // Fórmula automática para Risco Final var colRisco = 6 + (idx * 5); var cellGravRef = sheetMedias.getRange(linha_idx, colGrav).getA1Notation(); var cellProbRef = sheetMedias.getRange(linha_idx, colProb).getA1Notation(); sheetMedias.getRange(linha_idx, colRisco).setFormula( '=IF(OR(' + cellGravRef + '="—",' + cellProbRef + '=""),"Aguardando",' + 'IF(' + cellGravRef + '="Baixa",IF(' + cellProbRef + '="Alta","Médio","Baixo"),' + 'IF(' + cellGravRef + '="Média",IF(' + cellProbRef + '="Baixa","Baixo",IF(' + cellProbRef + '="Média","Médio","Alto")),' + 'IF(' + cellProbRef + '="Baixa","Médio",IF(' + cellProbRef + '="Média","Alto","Crítico")))))' ); }); linha_idx++; }); // Ajustar largura das colunas sheetMedias.autoResizeColumns(1, 3); sheetMedias.setFrozenRows(1); sheetMedias.setFrozenColumns(2); SpreadsheetApp.getUi().alert( '✅ Médias calculadas com sucesso!\n\n' + 'Total de grupos (CNPJ + Setor): ' + Object.keys(grupos).length + '\n\n' + 'PRÓXIMO PASSO:\n' + 'Preencha a coluna "Probabilidade" (Baixa/Média/Alta) para cada eixo.\n' + 'O Risco Final será calculado automaticamente.' ); }
